#include <iostream>
#include <fstream>

using namespace std;

typedef struct info_
{
    string modelo;
    string marca;
    string tipo;
    int ano;
    int quilometragem;
    float potencia;
    string combustivel;
    string cambio;
    string direcao;
    string cor;
    int portas;
    string placa;
}info;

typedef struct no_
{
    info * dados;
    no_ * prox;
}no;

typedef struct cabeca
{
    no_ * inicio;
    int tam;
    no_ * fim;
}cabecalho;
/**
 * @brief inicia uma lista com inicio, tamanho e fim
 */
cabecalho* inicia_lista () 
{
    cabecalho * tmp = new cabecalho;
    tmp->inicio = NULL;
    tmp->tam = 0;
    tmp->fim = NULL;
    return tmp;
}
/**
 * @brief Faz uma busca em uma lista ordenada
 *  
 * @param ptlista lista ordenada
 * @param placa parametro da busca
 * @param ant guarda o no anterior do no localizado ou do ultimo no menor que a placa
 * @param pont guarda o no caso o valor seja encontrado na lista
 */
void busca_lista(cabecalho * ptlista, string placa, no*ant, no*pont) 
{
    no * ptr = ptlista->inicio;
    *pont = NULL;
    *ant = ptlista->inicio;
    while (ptr != NULL) {
        if (ptr->dados->placa < placa) {
            *ant = ptr;
            ptr = ptr->prox;
        } else {
            if (ptr->dados->placa == placa) {
                *pont = ptr;
            } 
            ptr = NULL;
        }
    }
}
/**
 * @brief insere um nó no inicio da lista
 * 
 * @param ptlista lista para inserir
 * @param dados dados do nó que vai ser inserido
 */
int insere_inicio(cabecalho * ptlista, info * dados)
{
    no * tmp = new no;

    tmp->dados = dados;

    if (ptlista->tam == 0) {
        tmp->prox = NULL;
        ptlista->inicio = tmp;
        ptlista->fim = tmp;
    } else {
        tmp->prox = ptlista->inicio;
        ptlista->inicio = tmp;
    }
    ptlista->tam++;
}
/**
 * @brief insere um nó no fim da lista
 * 
 * @param ptlista lista para inserir
 * @param dados dados do nó que vai ser inserido
 */
int insere_fim(Pilha* ptlista, Carro* c)
{
    no* novoNo = new No;

    novoNo->carro = c;

    if (ptlista->tamanho == 0) {
        novoNo->proxNo = NULL;
        ptlista->cabeca = novoNo;
        ptlista->cauda = novoNo;
    } else {
        novoNo->proxNo = NULL;
        (ptlista->cauda)->proxNo = novoNo;
        ptlista->cauda = novoNo;
    }
    ptlista->tamanho++;
}
/**
 * @brief insere um nó em uma lista ordenada, utiliza a insere no inicio/fim para evitar que em todos os casos tenha que andar por todos os elementos da lista
 * 
 * @param ptlista lista para inserir
 * @param dados dados do nó que vai ser inserido
 * @return int 0 sucesso e -1 falha
 */
int insere_lista(cabecalho * ptlista, info * dados) 
{
    no * ant;
    no * pont;

    int retorno = -1;

    if (ptlista->inicio == NULL || ptlista->inicio->dados->placa > dados->placa) {
        insere_inicio(ptlista, dados);
        retorno = 0;
    } else if (ptlista->fim->dados->placa < dados->placa) {
        insere_fim(ptlista, dados);
        retorno = 0;
    } else {
        busca_lista(ptlista, dados->placa, &ant, &pont);

        if (pont == NULL) {
            no * tmp = new no;
            tmp->dados = dados;
            tmp->prox = ant->prox;
            ant->prox = tmp;
            ptlista->tam++;   
            retorno = 0; 
        }    
    }
    return retorno;
}
/**
 * @brief remove o nó do inicio da lista
 * 
 * @param ptlista lista para remoção
 * @return no* nó removido
 */
no * remove_inicio(cabecalho * ptlista)
{
    if (ptlista->tam == 1) {
        no * tmp = ptlista->inicio;
        ptlista->inicio = NULL;
        ptlista->fim = NULL;
        delete tmp;
    } else if (ptlista->tam > 1) {
        no * tmp = ptlista->inicio;
        ptlista->inicio = tmp->prox;
        delete tmp; 
    }
    ptlista->tam--;
}
/**
 * @brief remove o nó do fim da lista
 * 
 * @param ptlista lista para remoção
 * @return no* nó removido
 */
no * remove_fim(cabecalho * ptlista)
{
    if (ptlista->tam == 1) {
        no * tmp = ptlista->inicio;
        ptlista->inicio = NULL;
        ptlista->fim = NULL;
        delete tmp;
    } else if (ptlista->tam > 1) {
        no * tmp = ptlista->inicio;
        while (tmp->prox != ptlista->fim) {
            tmp = tmp->prox;
        }
        delete tmp->prox; 
        tmp->prox = NULL;
        ptlista->fim = tmp;
    }
    ptlista->tam--;
}
/**
 * @brief remove um nó de uma lista ordenada, utiliza a remove no inicio/fim para evitar que em todos os casos tenha que andar por todos os elementos da lista
 * 
 * @param ptlista lista para a remoção
 * @param placa placa do nó a ser removido
 * @return info* retorna o nó removido
 */
info * remove_lista(cabecalho * ptlista, string placa) 
{
    no * ant;
    no * pont;

    info * recuperado = NULL;

    if (ptlista->inicio->dados->placa == placa) {
        recuperado = ptlista->inicio->dados;
        remove_inicio(ptlista);
    } else if (ptlista->fim->dados->placa == placa) {
        recuperado = ptlista->fim->dados;
        remove_fim(ptlista);
    } else {
        busca_lista(ptlista, placa, &ant, &pont);  

        if (pont != NULL) {
            ant->prox = pont->prox;
            recuperado = pont->dados;
            ptlista->tam--;
            delete pont;
        }  
    }    
    return recuperado;
}

/**
 * @brief libera todos os nós de uma lista e volta o cabeçalho ao estado inicial 
 * 
 * @param ptlista lista para ser liberada
 */
void encerra_lista(cabecalho * ptlista)
{
    no * ant;
    no * ptr;
    if (ptlista->inicio != NULL) {
        ant = ptlista->inicio;
        ptr = (ptlista->inicio)->prox;
        while (ptr != NULL) {
            delete ant;
            ant = ptr;
            ptr = ptr->prox;
        }
        delete ant;     
        ptlista->tam = 0;
        ptlista->inicio = NULL;
        ptlista->fim = NULL;           
    } else {
        delete ptlista;
    }
}

/**
 * @brief faz a busca de um determinado valor escolhido pelo usuario, se a lista da busca estiver com valores ela é encerrada
 * 
 * @param ptlista lista principal

 * @param tipo tipo da busca, 1 para pilha 0 para fila
 * @param operador_busca operador que determina qual aspecto do carro vai ser buscado
 * @param param_busca_int parametro para valores inteiros
 * @param param_busca_float parametro para valores quebrados
 * @param param_busca_string parametro para palavras
 */
void new_buscas(cabecalho * ptlista, cabecalho * busca, int tipo, int operador_busca, int param_busca_int, float param_busca_float, string param_busca_string) 
{
    int aux;
    no * tmp = ptlista->inicio;

    if (busca->inicio != NULL) {
        encerra_lista(busca);
    }

    while (tmp != NULL) {
        aux = verifica_busca(tmp->dados, operador_busca, param_busca_int, param_busca_int, param_busca_string);
        if (aux == 0) {
            if (tipo == 1) {
                insere_inicio(busca, tmp->dados);    
            } else {
                insere_fim(busca, tmp->dados);
            } 
        }
        tmp = tmp->prox;
    }
}

/**
 * @brief dada uma string, converte ela para primeira letra maiuscula e o restante minusculo
 * 
 * @param palavra string para converter
 * @return string convertida
 */
string altera_string(string palavra)
{
    for (int i = 0; i < palavra.length(); i++) 
    {
        palavra[i] = tolower(palavra[i]);
    }
    palavra[0] = toupper(palavra[0]);
    return palavra;
}
